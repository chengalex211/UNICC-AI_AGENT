与quick_start_rag.py集成
═══════════════════════════════════════════════════════════════════

您现有的 quick_start_rag.py 可以无缝集成这些新文档:

修改 _load_documents() 方法:

```python
def _load_documents(self) -> List:
    """加载文档"""
    all_docs = []

    # 1. 加载 EU AI Act
    ai_act_dir = self.data_dir / "Markdown_for_RAG"
    if ai_act_dir.exists():
        loader = DirectoryLoader(str(ai_act_dir), glob="*.md")
        docs = loader.load()
        for doc in docs:
            doc.metadata["collection"] = "EU AI Act"
            doc.metadata["regulation"] = "EU AI Act"
        all_docs.extend(docs)
        print(f"   • EU AI Act: {len(docs)} 文件")

    # 2. 加载 GDPR (精简版)
    gdpr_dir = self.data_dir / "GDPR_AI_Focused"
    if gdpr_dir.exists():
        loader = DirectoryLoader(str(gdpr_dir), glob="P0_*.md")
        docs = loader.load()
        for doc in docs:
            doc.metadata["collection"] = "GDPR"
            doc.metadata["regulation"] = "GDPR"
        all_docs.extend(docs)
        print(f"   • GDPR AI-Focused: {len(docs)} 文件")

    # 3. 加载 Standards & Guidelines (新增)
    standards_dir = self.data_dir / "Standards_Guidelines_Markdown"
    if standards_dir.exists():
        loader = DirectoryLoader(
            str(standards_dir),
            glob="*.md",
            exclude=["README.md", "转换完成报告.txt"]
        )
        docs = loader.load()
        for doc in docs:
            doc.metadata["collection"] = "Standards & Guidelines"
            # 根据文件名设置分类
            if "NIST" in doc.metadata["source"]:
                doc.metadata["publisher"] = "NIST"
            elif "OWASP" in doc.metadata["source"]:
                doc.metadata["publisher"] = "OWASP"
            elif "UN" in doc.metadata["source"]:
                doc.metadata["publisher"] = "UN"
            elif "UNESCO" in doc.metadata["source"]:
                doc.metadata["publisher"] = "UNESCO"
        all_docs.extend(docs)
        print(f"   • Standards & Guidelines: {len(docs)} 文件")

    return all_docs
```
